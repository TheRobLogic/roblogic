# ABC License Tracker Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a daily-updating page at roblogic.org/jordon showing all newly issued California ABC licenses in San Diego County YTD, with a Leaflet map and filterable table.

**Architecture:** n8n workflow (daily cron) downloads CA ABC's bulk CSV, filters to SD County + current year + licensed, geocodes new addresses via Nominatim, outputs JSON. Static HTML page consumes JSON client-side. Data committed to GitHub repo via API, auto-deployed by Cloudflare Pages.

**Tech Stack:** n8n (workflow), vanilla HTML/CSS/JS (frontend), Leaflet.js + OpenStreetMap (map), Nominatim (geocoding), GitHub Contents API (deployment)

**Spec:** `docs/superpowers/specs/2026-03-12-abc-license-tracker-design.md`

```python
"""One-time seed: parse ABC CSV into jordon-abc-data.json"""
import csv
import json
from datetime import datetime, timezone

# License type labels and categories
TYPE_INFO = {
    2: ("Winegrower", "wholesale"),
    9: ("Beer & Wine Importer", "wholesale"),
    12: ("Distilled Spirits Importer", "wholesale"),
    17: ("Beer & Wine Wholesaler", "wholesale"),
    18: ("Distilled Spirits Wholesaler", "wholesale"),
    20: ("Off-Sale Beer & Wine", "off-premise"),
    21: ("Off-Sale General", "off-premise"),
    23: ("Small Beer Manufacturer", "wholesale"),
    40: ("On-Sale Beer", "on-premise"),
    41: ("On-Sale Beer & Wine — Eating Place", "on-premise"),
    42: ("On-Sale Beer & Wine — Public Premises", "on-premise"),
    47: ("On-Sale General — Eating Place", "on-premise"),
    48: ("On-Sale General — Public Premises", "on-premise"),
    58: ("Caterer's Permit", "on-premise"),
    61: ("On-Sale Beer — Public Premises", "on-premise"),
    68: ("Portable Bar", "on-premise"),
    70: ("On-Sale General — Restrictive Service", "on-premise"),
    75: ("Brewpub-Restaurant", "on-premise"),
    76: ("On-Sale General for Seasonal Business", "on-premise"),
    77: ("Event Permit", "other"),
    86: ("Instructional Tasting License", "other"),
}

current_year = str(datetime.now().year)
licenses = []

with open("/tmp/ABC-DailyDataExport.csv", "r", encoding="utf-8-sig") as f:
    # Skip metadata row
    next(f)
    reader = csv.DictReader(f)
    # Trim header names (some have leading spaces)
    reader.fieldnames = [name.strip() for name in reader.fieldnames]

    for row in reader:
        county = (row.get("Prem County") or "").strip().upper()
        lic_or_app = (row.get("Lic or App") or "").strip().upper()
        issue_date_raw = (row.get("Type Orig Iss Date") or "").strip()

        if county != "SAN DIEGO":
            continue
        if lic_or_app != "LIC":
            continue
        if current_year not in issue_date_raw:
            continue

        # Parse issue date from DD-MMM-YYYY to YYYY-MM-DD
        try:
            issue_dt = datetime.strptime(issue_date_raw, "%d-%b-%Y")
            issue_date = issue_dt.strftime("%Y-%m-%d")
        except ValueError:
            continue

        # Parse expiration date
        expir_raw = (row.get("Expir Date") or "").strip()
        try:
            expir_dt = datetime.strptime(expir_raw, "%d-%b-%Y")
            expir_date = expir_dt.strftime("%Y-%m-%d")
        except ValueError:
            expir_date = None

        license_type = int(row.get("License Type", 0))
        type_label, category = TYPE_INFO.get(license_type, (f"Type {license_type}", "other"))

        dba = (row.get("DBA Name") or "").strip()
        primary = (row.get("Primary Name") or "").strip()
        display_name = dba if dba else primary

        licenses.append({
            "fileNumber": (row.get("File Number") or "").strip(),
            "licenseType": license_type,
            "typeLabel": type_label,
            "category": category,
            "status": (row.get("Type Status") or "").strip(),
            "issueDate": issue_date,
            "expirationDate": expir_date,
            "businessName": primary,
            "dbaName": dba if dba else None,
            "displayName": display_name,
            "address": (row.get("Prem Addr 1") or "").strip(),
            "address2": (row.get("Prem Addr 2") or "").strip() or None,
            "city": (row.get("Prem City") or "").strip(),
            "zip": (row.get("Prem Zip") or "").strip(),
            "lat": None,
            "lng": None,
        })

# Sort by issue date descending
licenses.sort(key=lambda x: x["issueDate"], reverse=True)

output = {
    "updated": datetime.now(timezone.utc).astimezone().isoformat(),
    "count": len(licenses),
    "licenses": licenses,
}

with open("/Users/roblogic/github/roblogic/jordon/jordon-abc-data.json", "w") as f:
    json.dump(output, f, indent=2)

print(f"Wrote {len(licenses)} licenses to jordon-abc-data.json")
```

```javascript
const binaryKey = Object.keys(binaryData).find(k => k.endsWith('.csv')) || Object.keys(binaryData)[0];

function parseDate(raw) {
  // DD-MMM-YYYY → YYYY-MM-DD
  const months = {JAN:'01',FEB:'02',MAR:'03',APR:'04',MAY:'05',JUN:'06',
                   JUL:'07',AUG:'08',SEP:'09',OCT:'10',NOV:'11',DEC:'12'};
  const parts = raw.trim().split('-');
  if (parts.length !== 3) return null;
  const mon = months[parts[1].toUpperCase()];
  if (!mon) return null;
  return `${parts[2]}-${mon}-${parts[0].padStart(2, '0')}`;
}

function parseCSVLine(line) {
  const result = [];
  let current = '';
  let inQuotes = false;
  for (let i = 0; i < line.length; i++) {
    const ch = line[i];
    if (ch === '"') {
      if (inQuotes && i + 1 < line.length && line[i + 1] === '"') {
        current += '"'; // escaped quote
        i++;
      } else {
        inQuotes = !inQuotes;
      }
    } else if (ch === ',' && !inQuotes) {
      result.push(current);
      current = '';
    } else {
      current += ch;
    }
  }
  result.push(current);
  return result;
}

const licenses = [];

for (let i = 2; i < lines.length; i++) {
  const line = lines[i].trim();
  if (!line) continue;

  const fields = parseCSVLine(line);
  const row = {};
  headers.forEach((h, idx) => { row[h] = (fields[idx] || '').trim(); });

  const county = (row['Prem County'] || '').toUpperCase();
  const licOrApp = (row['Lic or App'] || '').toUpperCase();

  if (county !== 'SAN DIEGO') continue;
  if (licOrApp !== 'LIC') continue;

  const issueDate = parseDate(issueDateRaw);
  if (!issueDate) continue;

  const expirDate = parseDate(row['Expir Date'] || '');
  const licType = parseInt(row['License Type']) || 0;
  const [typeLabel, category] = TYPE_INFO[licType] || [`Type ${licType}`, 'other'];

  const dba = (row['DBA Name'] || '').trim();
  const primary = (row['Primary Name'] || '').trim();

  licenses.push({
    fileNumber: (row['File Number'] || '').trim(),
    licenseType: licType,
    typeLabel,
    category,
    status: (row['Type Status'] || '').trim(),
    issueDate,
    expirationDate: expirDate,
    businessName: primary,
    dbaName: dba || null,
    displayName: dba || primary,
    address: (row['Prem Addr 1'] || '').trim(),
    address2: (row['Prem Addr 2'] || '').trim() || null,
    city: (row['Prem City'] || '').trim(),
    zip: (row['Prem Zip'] || '').trim(),
    lat: null,
    lng: null,
  });
}

// Sort newest first
licenses.sort((a, b) => b.issueDate.localeCompare(a.issueDate));

return [{
  json: {
    updated: new Date().toISOString(),
    count: licenses.length,
    licenses,
  }
}];
```

### Task 8: Build the n8n workflow — geocode + merge + save

This task replaces the original SplitInBatches approach with a single Code node that handles geocoding sequentially via `https` (available in `NODE_FUNCTION_ALLOW_BUILTIN`). This is more reliable than SplitInBatches, which doesn't accumulate results across loop iterations.

- [ ] **Step 1: Add Code node (Geocode + Merge + Save)**

Node name: "Geocode and Save"

Connect: Parse and Filter SD County → Geocode and Save

This single node does everything: reads previous data, diffs for new addresses, geocodes them with rate limiting, merges results, saves to disk, and outputs base64 JSON for GitHub.

```javascript
const fs = require('fs');
const https = require('https');

const inputData = $input.first().json;
const licenses = inputData.licenses;

// --- Read previous data for diffing ---
let previousMap = {};
const dataPath = '/home/node/.n8n/abc-tracker-previous.json';
try {
  const prev = JSON.parse(fs.readFileSync(dataPath, 'utf-8'));
  for (const lic of prev.licenses) {
    previousMap[lic.fileNumber] = lic;
  }
} catch (e) {
  // First run or file missing — geocode everything
}

// --- Diff: reuse cached geocodes, identify what needs geocoding ---
const needsGeocoding = [];
for (const lic of licenses) {
  const prev = previousMap[lic.fileNumber];
  if (prev && prev.lat !== null && prev.lat !== undefined) {
    lic.lat = prev.lat;
    lic.lng = prev.lng;
  } else {
    needsGeocoding.push(lic);
  }
}

// --- Geocode helper: returns a Promise that resolves after HTTP GET ---
function geocodeAddress(address, city, zip) {
  return new Promise((resolve) => {
    const params = new URLSearchParams({
      street: address,
      city: city,
      state: 'California',
      postalcode: zip,
      format: 'json',
      limit: '1',
    });
    const url = `https://nominatim.openstreetmap.org/search?${params}`;

    const req = https.get(url, {
      headers: { 'User-Agent': 'ABCLicenseTracker/1.0 (roblogic.org)' }
    }, (res) => {
      let body = '';
      res.on('data', chunk => body += chunk);
      res.on('end', () => {
        try {
          const data = JSON.parse(body);
          if (data.length > 0) {
            resolve({ lat: parseFloat(data[0].lat), lng: parseFloat(data[0].lon) });
          } else {
            resolve(null);
          }
        } catch (e) {
          resolve(null);
        }
      });
    });
    req.on('error', () => resolve(null));
    req.end();
  });
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

// --- Geocode new addresses sequentially with 1s rate limit ---
for (const lic of needsGeocoding) {
  const result = await geocodeAddress(lic.address, lic.city, lic.zip);
  if (result) {
    lic.lat = result.lat;
    lic.lng = result.lng;
  }
  await sleep(1100); // 1.1s to stay well within Nominatim's 1 req/sec limit
}

// --- Build final output ---
const output = {
  updated: new Date().toISOString(),
  count: licenses.length,
  licenses,
};

// Save to disk for next run's diff
fs.writeFileSync(dataPath, JSON.stringify(output, null, 2));

// Output base64-encoded JSON for GitHub API commit
const jsonStr = JSON.stringify(output, null, 2);
return [{
  json: {
    content: Buffer.from(jsonStr).toString('base64'),
    count: output.count,
    geocoded: needsGeocoding.length,
    cached: licenses.length - needsGeocoding.length,
  }
}];
```

```javascript
// --- Helper: HTTPS request as Promise ---
function apiRequest(method, url, body, authToken) {
  return new Promise((resolve, reject) => {
    const urlObj = new URL(url);
    const options = {
      hostname: urlObj.hostname,
      path: urlObj.pathname + urlObj.search,
      method,
      headers: {
        'User-Agent': 'ABCLicenseTracker/1.0',
        'Authorization': `Bearer ${authToken}`,
        'Accept': 'application/vnd.github+json',
      },
    };
    if (body) {
      const bodyStr = JSON.stringify(body);
      options.headers['Content-Type'] = 'application/json';
      options.headers['Content-Length'] = Buffer.byteLength(bodyStr);
    }

    const req = https.request(options, (res) => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        try {
          resolve({ status: res.statusCode, data: JSON.parse(data) });
        } catch (e) {
          resolve({ status: res.statusCode, data });
        }
      });
    });
    req.on('error', reject);
    if (body) req.write(JSON.stringify(body));
    req.end();
  });
}

// --- Step 1: Get current file SHA (if it exists) ---
let sha = null;
try {
  const getResult = await apiRequest('GET', API_BASE, null, token);
  if (getResult.status === 200 && getResult.data.sha) {
    sha = getResult.data.sha;
  }
  // 404 = file doesn't exist yet, that's fine
} catch (e) {
  // Network error — will fail on PUT too, let it propagate
}

// --- Step 2: Build commit body ---
const today = new Date().toISOString().slice(0, 10);
const commitBody = {
  message: `auto: abc license update ${today}`,
  content,
};
if (sha) {
  commitBody.sha = sha;
}

// --- Step 3: PUT to create/update file ---
const putResult = await apiRequest('PUT', API_BASE, commitBody, token);

if (putResult.status !== 200 && putResult.status !== 201) {
  throw new Error(`GitHub API error ${putResult.status}: ${JSON.stringify(putResult.data)}`);
}

return [{
  json: {
    success: true,
    sha: putResult.data.content?.sha,
    commitUrl: putResult.data.commit?.html_url,
  }
}];
```
